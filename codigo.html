Ejemplo de Código JavaScript para Sumar


// Función para sumar dos números
function sumar(a, b) {
    return a + b;
}

// Se agregará el evento para el botón después de cargar el DOM
document.getElementById('sumButton').addEventListener('click', function() {
    // Obtiene los valores de los inputs y los convierte a número
    const a = Number(document.getElementById('a').value);
    const b = Number(document.getElementById('b').value);

    // Usa la función sumar para calcular la suma
    const resultado = sumar(a, b);

    // Muestra el resultado en el párrafo con id "output"
    document.getElementById('output').textContent = 'Resultado: ' + resultado;
});
